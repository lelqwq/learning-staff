void shellSort(int arr[],int n)//n为arr数组的大小 
{
    for (int gap = n/2; gap > 0; gap /= 2)//循环增量 
    {
        for (int i = gap; i < n; i += 1)//对子数组进行选择排序
	{
            int temp = arr[i];
            int j;
            for (j = i; j >= gap && arr[j - gap] > temp; j -= gap)
	    {
                arr[j] = arr[j - gap];
 	    }
            arr[j] = temp;
        }
    }
}
